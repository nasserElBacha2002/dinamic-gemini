```py
# src/application/use_cases/upload_aisle_assets.py
from uuid import uuid4
from src.domain.assets.entities import SourceAsset, SourceAssetType


class UploadAisleAssetsUseCase:
    def __init__(self, aisle_repo, asset_repo, artifact_storage, clock):
        self.aisle_repo = aisle_repo
        self.asset_repo = asset_repo
        self.artifact_storage = artifact_storage
        self.clock = clock

    def execute(self, aisle_id: str, files: list):
        aisle = self.aisle_repo.get_by_id(aisle_id)
        if aisle is None:
            raise ValueError("Aisle not found")

        now = self.clock.now()
        created_assets = []

        for file in files:
            asset_id = str(uuid4())
            storage_path = f"aisles/{aisle_id}/raw/{asset_id}_{file.filename}"
            final_path = self.artifact_storage.save_file(
                path=storage_path,
                file_obj=file.file,
                content_type=file.content_type,
            )

            asset_type = SourceAssetType.VIDEO if file.content_type.startswith("video/") else SourceAssetType.PHOTO
            asset = SourceAsset(
                id=asset_id,
                aisle_id=aisle_id,
                type=asset_type,
                original_filename=file.filename,
                storage_path=final_path,
                mime_type=file.content_type,
                uploaded_at=now,
            )
            self.asset_repo.save(asset)
            created_assets.append(asset)

        aisle.mark_assets_uploaded(now)
        self.aisle_repo.save(aisle)
        return created_assets
```

```py
class EnqueueAisleProcessingUseCase:
    def __init__(self, aisle_repo, job_repo, job_queue, clock):
        self.aisle_repo = aisle_repo
        self.job_repo = job_repo
        self.job_queue = job_queue
        self.clock = clock

    def execute(self, aisle_id: str) -> str:
        aisle = self.aisle_repo.get_by_id(aisle_id)
        if aisle is None:
            raise ValueError("Aisle not found")

        now = self.clock.now()
        queue_job_id = self.job_queue.enqueue(
            job_type="process_aisle",
            payload={"aisle_id": aisle_id},
        )

        job = Job(
            id=queue_job_id,
            target_type="aisle",
            target_id=aisle_id,
            job_type="process_aisle",
            status=JobStatus.QUEUED,
            created_at=now,
            updated_at=now,
        )
        self.job_repo.save(job)

        aisle.status = AisleStatus.QUEUED
        aisle.updated_at = now
        self.aisle_repo.save(aisle)
        return job.id
```

## **HU-9.1 — Como responsable, quiero consultar métricas de inventario para evaluar desempeño y calidad**

### **Criterios de aceptación**

El cálculo debe seguir la **definición canónica de métricas** del **Documento técnico 3.0 §9.6** (total_reviewed_positions, auto_accepted_positions, success_rate, correction_rate, deletion_rate).

### **Tareas técnicas**

* Crear contrato `MetricsCalculator`.  
* Implementar servicio de cálculo.  
* Evitar lógica de métricas dentro de controladores o repositorios.

### **Código sugerido**

```py
class MetricsCalculator(ABC):
    @abstractmethod
    def calculate_inventory_metrics(self, inventory_id: str) -> dict:
        raise NotImplementedError
```

```py
class InventoryMetricsService(MetricsCalculator):
    def __init__(self, aisle_repo, position_repo):
        self.aisle_repo = aisle_repo
        self.position_repo = position_repo

    def calculate_inventory_metrics(self, inventory_id: str) -> dict:
        aisles = self.aisle_repo.list_by_inventory(inventory_id)
        aisle_ids = [a.id for a in aisles]
        positions = self.position_repo.list_by_aisles(aisle_ids)

        total = len(positions)
        # Solo posiciones con estado terminal (reviewed, corrected, deleted) — ver Documento técnico 3.0 §9.6
        reviewed_positions = [p for p in positions if p.status.value in ("reviewed", "corrected", "deleted")]
        total_reviewed = len(reviewed_positions)
        corrected = sum(1 for p in reviewed_positions if p.status.value == "corrected")
        deleted = sum(1 for p in reviewed_positions if p.status.value == "deleted")
        auto_accepted = sum(1 for p in reviewed_positions if p.status.value == "reviewed")

        success_rate = (auto_accepted / total_reviewed * 100) if total_reviewed else 0.0

        return {
            "total_reviewed_positions": total_reviewed,
            "total_positions": len(positions),
            "corrected_positions": corrected,
            "deleted_positions": deleted,
            "auto_accepted_positions": auto_accepted,
            "success_rate": round(success_rate, 2),
        }
```

# **Reglas técnicas para respetar SOLID durante la implementación**

## **1\. Los controladores HTTP no deben contener lógica de negocio**

Solo:

* parsean request,  
* llaman use case,  
* serializan response.

## **2\. Los casos de uso no deben conocer SQL ni filesystem**

Solo dependen de interfaces.

## **3\. Las entidades no deben depender del framework**

Nada de:

* decoradores de ORM,  
* objetos request/response,  
* imports del proveedor LLM.

## **4\. No crear una clase “manager” gigante**

Evitar algo tipo:

* `InventoryPipelineManager`  
  que haga todo.

Separar:

* upload,  
* enqueue,  
* process,  
* map results,  
* review,  
* metrics.

## **5\. Los DTOs de API no son las entidades del dominio**

Separar claramente:

* request/response schema,  
* command/query,  
* entity.
